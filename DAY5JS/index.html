Q1. Create a button , on click of which new Heading tag h1 should be added with text as "MERN stack" on the screen above button

        Ans:
        var Heading =document.getElementsByClassName("Heading");
var count=0;
function  change(){
    if(count%2==0){
        Heading[0].innerHTML="MERN stack";
        count++;
    }else{
        Heading[0].innerHTML="";
        count--;
    }
   
}

Q3. Write code to implement timer clock using JS -- display HH:MM:SS -- the time should keep updating every second

Start clock

        Ans:
        var head2=document.getElementsByTagName("h2");

        function clock(){
            var time=new Date().toLocaleTimeString();
        head2[0].innerHTML=time
        setInterval(() => {
            clock()
        }, 1000);
        }
        
        head2[0].addEventListener("click",clock );
